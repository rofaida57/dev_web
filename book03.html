2. تعريف المتغيرات

4. الشروط (if - else)

if (score >= 90) {
    console.log("ممتاز");
} else if (score >= 70) {
    console.log("جيد");
} else {
    console.log("ضعيف");
}

5. الحلقات التكرارية

for (let i = 0; i < 5; i++) {
    console.log(i);
}

6. إنشاء دالة

function greet(name) {
    return "مرحبًا " + name;
}

7. استخدام الدوال السهمية

const sum = (a, b) => a + b;

8. التعامل مع المصفوفات

numbers.forEach(num => console.log(num));
let doubled = numbers.map(num => num * 2);

9. التعامل مع الكائنات

let person = {
    name: "Ali",
    age: 25,
    greet() {
        console.log("مرحبًا، اسمي " + this.name);
    }
};

10. التحقق من القيم

console.log(5 == "5");  // true
console.log(5 === "5"); // false

11. التعامل مع الوقت والتاريخ

let now = new Date();
console.log(now.toLocaleString());

12. استخدام setTimeout و setInterval

setTimeout(() => console.log("تم التنفيذ"), 2000);
setInterval(() => console.log("تكرار"), 1000);

13. استخدام try...catch

try {
    let result = x / 0;
} catch (error) {
    console.log("حدث خطأ: " + error.message);
}

14. جلب بيانات من API

fetch("https://jsonplaceholder.typicode.com/posts/1")
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.log("خطأ: " + error));

16. التعامل مع الأحداث

document.getElementById("btn").addEventListener("click", () => {

});

17. استخدام localStorage

localStorage.setItem("username", "Ali");
console.log(localStorage.getItem("username"));
