أساسيات Java

1. طباعة النصوص

System.out.println("مرحبًا بالعالم!");

2. المتغيرات

int age = 25; double pi = 3.14; String name = "أحمد"; boolean isAdult = true;

3. العمليات الحسابية

int sum = 10 + 5; int product = 10 * 5; double division = 10.0 / 2;

4. الشروط

if (age > 18) { System.out.println("بالغ"); } else { System.out.println("قاصر"); }

5. الحلقات

for (int i = 0; i < 5; i++) { System.out.println(i); } while (age < 30) { age++; }

6. المصفوفات

int[] numbers = {1, 2, 3, 4, 5}; System.out.println(numbers[0]); // 1

7. الكائنات

class Person { String name; int age; } Person p = new Person(); p.name = "أحمد";

8. الدوال

public static int add(int a, int b) { return a + b; }

9. التعامل مع النصوص

String text = "مرحبًا".toUpperCase(); System.out.println(text); // مرحبًا بحروف كبيرة

10. إنشاء كائنات

Person person = new Person(); person.name = "محمد"; person.age = 30;
