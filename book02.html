أساسيات CSS

جرب البحث بكلمات أخرى

الألوان والخلفيات

color: red; /* لون النص */
background-color: blue; /* لون الخلفية */
background-image: url('image.jpg'); /* صورة خلفية */
background-repeat: no-repeat; /* منع تكرار الخلفية */
background-size: cover; /* تغطية الخلفية */
background-position: center; /* محاذاة الخلفية */
background-attachment: fixed; /* جعل الخلفية ثابتة عند التمرير */

الخطوط والنصوص

font-size: 16px; /* حجم الخط */
font-family: Arial, sans-serif; /* نوع الخط */
font-weight: bold; /* خط عريض */
font-style: italic; /* خط مائل */
text-align: center; /* محاذاة النص */
text-decoration: underline; /* خط تحت النص */
text-transform: uppercase; /* تحويل النص إلى أحرف كبيرة */
letter-spacing: 2px; /* تباعد الأحرف */
line-height: 1.5; /* ارتفاع السطر */
word-spacing: 5px; /* تباعد الكلمات */

الهوامش والتباعد

margin: 10px; /* المسافة الخارجية */
margin-top: 20px; /* الهامش العلوي */
padding: 10px; /* المسافة الداخلية */

الحدود والظلال

border: 2px solid black; /* تحديد الحدود */
border-radius: 10px; /* تدوير الزوايا */
box-shadow: 5px 5px 10px rgba(0, 0, 0, 0.3); /* ظل */

التحكم في الأحجام

width: 100px; /* العرض */
height: 200px; /* الارتفاع */
max-width: 500px; /* الحد الأقصى للعرض */
min-height: 100px; /* الحد الأدنى للارتفاع */

العرض والتخطيط

display: flex; /* استخدام Flexbox */
justify-content: center; /* محاذاة العناصر أفقيًا */
align-items: center; /* محاذاة العناصر عموديًا */
flex-direction: column; /* تغيير اتجاه العناصر إلى عمودي */
gap: 10px; /* تباعد بين العناصر */
visibility: hidden; /* إخفاء العنصر */

التحولات والأنيميشن

transition: all 0.3s ease-in-out; /* حركة ناعمة */
transform: rotate(45deg); /* تدوير العنصر */
animation: fadeIn 2s infinite; /* تطبيق أنيميشن مكرر */

تحديد المواقع

position: absolute; /* تحديد الموقع */
top: 50px; /* المسافة من الأعلى */
left: 100px; /* المسافة من اليسار */
z-index: 10; /* ترتيب الطبقات */

التنسيقات التفاعلية

:hover { background-color: yellow; } /* عند مرور الماوس */
:focus { border: 2px solid blue; } /* عند تحديد العنصر */
@media (max-width: 600px) { body { background-color: lightgray; } } /* تصميم استجابة */

الجداول والقوائم

border-collapse: collapse; /* دمج حدود الجدول */
list-style-type: none; /* إزالة نقاط القائمة */

الأزرار والتفاعلات

cursor: pointer; /* تغيير شكل المؤشر عند المرور */
user-select: none; /* منع تحديد النص */
opacity: 0.5; /* شفافية العنصر */

التمرير

overflow: hidden; /* إخفاء التمرير */
overflow-y: scroll; /* إظهار التمرير العمودي */

الشفافية والفلاتر

opacity: 0.7; /* مستوى الشفافية */
filter: blur(5px); /* تأثير التمويه */
